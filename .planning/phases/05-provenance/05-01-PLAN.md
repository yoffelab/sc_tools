def _leiden_cluster(X: np.ndarray, resolution: float = 1.0) -> np.ndarray:
    """Run Leiden clustering on an embedding."""
    import scanpy as sc
    tmp = AnnData(X=np.zeros((X.shape[0], 1)))
    tmp.obsm["X_emb"] = X
    sc.pp.neighbors(tmp, use_rep="X_emb", n_neighbors=15)
    sc.tl.leiden(tmp, resolution=resolution, key_added="leiden")
    return tmp.obs["leiden"].values
```

def leiden(adata, resolution=0.8, key_added="leiden", **kwargs):
    backend, name = get_backend()

def cluster(adata, resolution=0.8, use_rep=None, n_neighbors=20, key_added="leiden", run_umap=True, **kwargs):
    neighbors(adata, n_neighbors=n_neighbors, use_rep=use_rep, **kwargs)
    leiden(adata, resolution=resolution, key_added=key_added)
    if run_umap: umap(adata)
```
