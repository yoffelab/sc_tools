    - Pattern:
      ```python
      def cli_handler(func=None, *, tier=None):
          def decorator(fn):
              @functools.wraps(fn)
              def wrapper(*args, **kwargs):
                  dry_run = kwargs.pop("dry_run", False)
                  force = kwargs.pop("force", False)
                  start_time = time.monotonic()

                  if dry_run:
                      # Build dry-run result
                      file_arg = kwargs.get("file") or kwargs.get("input_file") or kwargs.get("from_dir")
                      dry_data = {"planned_command": fn.__name__, "tier": tier.value if tier else "full"}
                      if file_arg and Path(file_arg).exists():
                          from sc_tools.io.estimate import estimate_from_h5
                          try:
                              est = estimate_from_h5(str(file_arg))
                              dry_data["estimate"] = est
                          except Exception:
                              dry_data["estimate"] = None
                      elif file_arg:
                          from sc_tools.errors import SCToolsUserError
                          raise SCToolsUserError(f"Input file not found: {file_arg}", suggestion="Check the file path")
                      result = CLIResult(status=Status.skipped, command=fn.__name__, data=dry_data, message="Dry run -- no data modified")
                      _emit(result)
                      raise SystemExit(0)

                  # Normal execution (existing logic)
                  try:
                      result = fn(*args, **kwargs)
                      # ... rest of existing handler
              return wrapper
          if func is not None:
              return decorator(func)
          return decorator
      ```
    - The `tier` parameter is stored but not used for automatic loading in this plan -- it is metadata for dry-run reporting and future Gateway integration. Commands still load data themselves.

    **Create `sc_tools/cli/estimate.py`:**
    - Function `register_estimate(app: typer.Typer) -> None` following register_discovery pattern.
    - Command: `sct estimate <command_name> <file>` where command_name is like "preprocess", "qc", "de", "benchmark".
    - Implementation:
      ```python
      def register_estimate(app: typer.Typer) -> None:
          @app.command("estimate")
          def estimate_command(
              command_name: str = typer.Argument(..., help="Command to estimate (e.g., preprocess, qc, de)"),
              file: str = typer.Argument(..., help="Input h5ad file path"),
          ) -> None:
              """Estimate peak memory and runtime for a command before execution."""
              from pathlib import Path
              from sc_tools.io.estimate import estimate_from_h5, METHOD_MULTIPLIERS
              from sc_tools.models.result import CLIResult, Status

              file_path = Path(file)
              if not file_path.exists():
                  from sc_tools.errors import SCToolsUserError
                  raise SCToolsUserError(f"File not found: {file}", suggestion="Check the file path")

              est = estimate_from_h5(str(file_path))

              # Apply method-specific multiplier
              method_key = command_name.split()[0]  # Handle "preprocess run" -> "preprocess"
              multiplier = METHOD_MULTIPLIERS.get(method_key, METHOD_MULTIPLIERS["default"])
              est["method_multiplier"] = multiplier
              est["estimated_peak_mb"] = est["base_mb"] * multiplier

              # Runtime estimate (rough): ~1 second per 10K cells for most operations
              throughput = {"preprocess": 5000, "de": 8000, "qc": 20000, "benchmark": 10000}
              cells_per_sec = throughput.get(method_key, 10000)
              est["estimated_runtime_s"] = round(est["n_obs"] / cells_per_sec, 1)
              est["runtime_confidence"] = "low"
              est["memory_confidence"] = "high"

              result = CLIResult(
                  status=Status.success,
                  command=f"estimate {command_name}",
                  data=est,
                  message=f"Estimated peak memory: {est['estimated_peak_mb']:.0f}MB for {est['n_obs']} cells x {est['n_vars']} genes",
              )
              # Use _emit from cli/__init__.py
              from sc_tools.cli import _emit
              _emit(result)
              raise SystemExit(0)
      ```
